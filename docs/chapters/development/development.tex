\chapter{Development}\label{ch:development}
\section{Setup the development environment}
\begin{enumerate}
    \item Clone the source: \ttt{git clone https://github.com/sakurawald/fuji-fabric.git}
    \item Change the working-directory: \ttt{cd fuji-fabric}
    \item Compile the source: \ttt{./gradlew build}
\end{enumerate}
