\LevelOne{multiplier}
\LevelTwo{Purpose}
This module provides some \tbf{numeric multiplier}.

\LevelTwo{Supported Numeric Types}
\begin{description}
    \item [damage] damage to player
    \item[experience] experience a player gained
\end{description}

\LevelTwo{Example}
\begin{example}{Double the damage from zombie to a player}
    \ic{/lp group default meta set fuji.multiplier.damage.minecraft:zombie 2}
\end{example}

\begin{example}{Cancel the fall damage}
    \ic{/lp group default meta set fuji.multiplier.damage.minecraft:fall 0}
\end{example}

\begin{example}{Double all damage to a player}
    \ic{/lp group default meta set fuji.multiplier.damage.all 2}
\end{example}

\begin{example}{Double all experience a player gained}
    \ic{/lp group default meta set fuji.multiplier.experience.all 2}
\end{example}

\begin{example}{Half all damage to a player}
    \ic{/lp group default meta set fuji.multiplier.damage.all 0.5}
\end{example}
