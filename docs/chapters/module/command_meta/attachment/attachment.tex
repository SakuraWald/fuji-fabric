\sss{attachment}

\ssss{Purpose}
Provides a unified attachment facility, which can attach any data to any object.

\ssss{Command}
\sssss{/attachment}

\ssss{Example}

\begin{example}{Set a attachment}
   \ic{/attachment set news today hello world}
\end{example}

\begin{example}{Get a attachment}
    \ic{/attachment get news today}
\end{example}
