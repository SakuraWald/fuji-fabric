% Define custom boxes
\tcbuselibrary{listings}

\newtcolorbox[auto counter, number within=section]{note}[2][]{%
    colback=blue!5!white, colframe=blue!75!black,
    fonttitle=\bfseries, title=Note: #2,
    sharp corners=south,
    boxrule=0.8mm,
    #1
}
